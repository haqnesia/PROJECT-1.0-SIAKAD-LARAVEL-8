    // AJAX TAMPIL

    function view_data() {
        $.get("{{ route('fakultas.data') }}")
            .done(function (r) {
                if (r.success) {
                    $('#data_view').html(r.html);
                } else {
                    alert(r.message);
                }
            })
            .fail(function (x) {
                alert(x.status + ' - ' + x.statusText);
            });
    }
